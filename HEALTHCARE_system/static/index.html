🐍 Python Triage Engine
# services/triage_engine.py def classify_patient(symptoms, temp, o2, age): RED_FLAGS = {"chest_pain", "unconscious", "difficulty_breathing", ...} for s in symptoms: if s in RED_FLAGS: return TriageResult("EMERGENCY", score=99) score = 0 if temp >= 104: score += 3 elif temp >= 102: score += 1 if o2 <= 90: score += 3 elif o2 <= 94: score += 2 if age in ("child", "senior"): score += 1 return "EMERGENCY" if score >= 4 else "NORMAL"
